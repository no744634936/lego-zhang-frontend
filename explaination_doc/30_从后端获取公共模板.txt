    import axios from "axios"

    axios.defaults.baseURL=`${baseBackendUrl}/api/`



store/templates.ts 里写上
mutations里的方法必须都是同步的，因为异步会破坏vuex的可追溯性
actions里面可以使用异步方法

    actions:{
        fetchTemplates(context){
            return axios.get('/templates').then(resp=>{
                context.commit("fetchTemplates",resp.data)  //触发fetchTemplates 这个mutation
            })
        }
    },

    mutations:{
        fetchTemplates(state,respData: RespListData<TemplateProps>){
            state.data=respData.data.list
        }
    }



Home.vue 里写上

    onMounted(()=>{
        store.dispatch("fetchTemplates") // 触发的是action里的fetchTemplates 方法
    })
